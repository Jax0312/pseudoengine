CLASS Pet INHERITS Snake
    PUBLIC Name : STRING
    PUBLIC Species: ARRAY[1:3] OF STRING
    PUBLIC PROCEDURE NEW(GivenName : STRING)
        Name <- GivenName
    ENDPROCEDURE
ENDCLASS

DECLARE pet: Pet
pet <- NEW Pet("Samuel")
pet.Species[5 - 4] <- "cobra"
OUTPUT pet.Name, " is a ", pet.Species[4 - 3]
