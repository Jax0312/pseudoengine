CLASS Snake
    PUBLIC Name : STRING
    PUBLIC Owners : ARRAY[1:3] OF STRING
    PUBLIC PROCEDURE NEW(GivenName : STRING)
        Name <- GivenName
        Owners[1] <- "James"
    ENDPROCEDURE

    PUBLIC PROCEDURE Rename(GivenName : STRING)
        Name <- GivenName
    ENDPROCEDURE
ENDCLASS

DECLARE snake: Snake
snake <- NEW Snake("Samuel")
OUTPUT snake.Name, " is a snake belonging to ", snake.Owners[1]
snake.Rename("Simon")
OUTPUT "It's new name is ", snake.Name
